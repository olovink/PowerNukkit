001package cn.nukkit.network.rcon;
002
003import java.nio.channels.SocketChannel;
004
005/**
006 * A data structure to hold sender, request ID and command itself.
007 *
008 * @author Tee7even
009 */
010public class RCONCommand {
011    private final SocketChannel sender;
012    private final int id;
013    private final String command;
014
015    public RCONCommand(SocketChannel sender, int id, String command) {
016        this.sender = sender;
017        this.id = id;
018        this.command = command;
019    }
020
021    public SocketChannel getSender() {
022        return this.sender;
023    }
024
025    public int getId() {
026        return this.id;
027    }
028
029    public String getCommand() {
030        return this.command;
031    }
032}
