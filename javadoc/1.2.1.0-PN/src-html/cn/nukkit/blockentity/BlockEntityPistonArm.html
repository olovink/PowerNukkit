001package cn.nukkit.blockentity;
002
003import cn.nukkit.Player;
004import cn.nukkit.block.Block;
005import cn.nukkit.block.BlockAir;
006import cn.nukkit.block.BlockID;
007import cn.nukkit.entity.Entity;
008import cn.nukkit.event.entity.EntityMoveByPistonEvent;
009import cn.nukkit.level.format.FullChunk;
010import cn.nukkit.math.AxisAlignedBB;
011import cn.nukkit.math.BlockFace;
012import cn.nukkit.math.BlockVector3;
013import cn.nukkit.math.SimpleAxisAlignedBB;
014import cn.nukkit.nbt.tag.CompoundTag;
015import cn.nukkit.nbt.tag.IntTag;
016import cn.nukkit.nbt.tag.ListTag;
017import cn.nukkit.utils.Faceable;
018
019import java.util.ArrayList;
020import java.util.List;
021
022/**
023 * @author CreeperFace
024 */
025public class BlockEntityPistonArm extends BlockEntitySpawnable {
026
027    public static final float MOVE_STEP = Float.valueOf(0.5f);
028
029    public float progress;
030    public float lastProgress = 1;
031    public BlockFace facing;
032    public boolean extending;
033    public boolean sticky;
034    public int state;
035    public int newState = 1;
036    public List<BlockVector3> attachedBlocks;
037    public boolean powered;
038
039    public BlockEntityPistonArm(FullChunk chunk, CompoundTag nbt) {
040        super(chunk, nbt);
041    }
042
043    @Override
044    protected void initBlockEntity() {
045        if (namedTag.contains("Progress")) {
046            this.progress = namedTag.getFloat("Progress");
047        }
048
049        if (namedTag.contains("LastProgress")) {
050            this.lastProgress = (float) namedTag.getInt("LastProgress");
051        }
052
053        this.sticky = namedTag.getBoolean("Sticky");
054        this.extending = namedTag.getBoolean("Extending");
055        this.powered = namedTag.getBoolean("powered");
056
057
058        if (namedTag.contains("facing")) {
059            this.facing = BlockFace.fromIndex(namedTag.getInt("facing"));
060        } else {
061            Block b = this.getLevelBlock();
062
063            if (b instanceof Faceable) {
064                this.facing = ((Faceable) b).getBlockFace();
065            } else {
066                this.facing = BlockFace.NORTH;
067            }
068        }
069
070        attachedBlocks = new ArrayList<>();
071
072        if (namedTag.contains("AttachedBlocks")) {
073            ListTag blocks = namedTag.getList("AttachedBlocks", IntTag.class);
074            if (blocks != null && blocks.size() > 0) {
075                for (int i = 0; i < blocks.size(); i += 3) {
076                    this.attachedBlocks.add(new BlockVector3(
077                            ((IntTag) blocks.get(i)).data,
078                            ((IntTag) blocks.get(i + 1)).data,
079                            ((IntTag) blocks.get(i + 1)).data
080                    ));
081                }
082            }
083        } else {
084            namedTag.putList(new ListTag<>("AttachedBlocks"));
085        }
086
087        super.initBlockEntity();
088    }
089
090    private void moveCollidedEntities() {
091        BlockFace pushDir = this.extending ? facing : facing.getOpposite();
092        for (BlockVector3 pos : this.attachedBlocks) {
093            BlockEntity blockEntity = this.level.getBlockEntity(pos.getSide(pushDir));
094
095            if (blockEntity instanceof BlockEntityMovingBlock) {
096                ((BlockEntityMovingBlock) blockEntity).moveCollidedEntities(this, pushDir);
097            }
098        }
099
100        AxisAlignedBB bb = new SimpleAxisAlignedBB(0, 0, 0, 1, 1, 1).getOffsetBoundingBox(
101                this.x + (pushDir.getXOffset() * progress),
102                this.y + (pushDir.getYOffset() * progress),
103                this.z + (pushDir.getZOffset() * progress)
104        );
105
106        Entity[] entities = this.level.getCollidingEntities(bb);
107
108        for (Entity entity : entities) {
109            moveEntity(entity, pushDir);
110        }
111    }
112
113    void moveEntity(Entity entity, BlockFace moveDirection) {
114        if (!entity.canBePushed()) {
115            return;
116        }
117
118        EntityMoveByPistonEvent event = new EntityMoveByPistonEvent(entity, entity.getPosition());
119        this.level.getServer().getPluginManager().callEvent(event);
120
121        if (event.isCancelled()) {
122            return;
123        }
124
125        if (entity instanceof Player) {
126            return;
127        }
128
129        entity.onPushByPiston(this);
130
131        if (!entity.closed) {
132            float diff = Math.abs(this.progress - this.lastProgress);
133
134            entity.move(
135                    diff * moveDirection.getXOffset(),
136                    diff * moveDirection.getYOffset(),
137                    diff * moveDirection.getZOffset()
138            );
139        }
140    }
141
142    public void move(boolean extending, List<BlockVector3> attachedBlocks) {
143        this.extending = extending;
144        this.lastProgress = this.progress = extending ? 0 : 1;
145        this.state = this.newState = extending ? 1 : 3;
146        this.attachedBlocks = attachedBlocks;
147        this.movable = false;
148
149        this.level.addChunkPacket(getChunkX(), getChunkZ(), getSpawnPacket());
150        this.lastProgress = extending ? -MOVE_STEP : 1 + MOVE_STEP;
151        this.moveCollidedEntities();
152        this.scheduleUpdate();
153    }
154
155    @Override
156    public boolean onUpdate() {
157        boolean hasUpdate = true;
158
159        if (this.extending) {
160            this.progress = Math.min(1, this.progress + MOVE_STEP);
161            this.lastProgress = Math.min(1, this.lastProgress + MOVE_STEP);
162        } else {
163            this.progress = Math.max(0, this.progress - MOVE_STEP);
164            this.lastProgress = Math.max(0, this.lastProgress - MOVE_STEP);
165        }
166
167        this.moveCollidedEntities();
168
169        if (this.progress == this.lastProgress) {
170            this.state = this.newState = extending ? 2 : 0;
171
172            BlockFace pushDir = this.extending ? facing : facing.getOpposite();
173
174            for (BlockVector3 pos : this.attachedBlocks) {
175                BlockEntity movingBlock = this.level.getBlockEntity(pos.getSide(pushDir));
176
177                if (movingBlock instanceof BlockEntityMovingBlock) {
178                    movingBlock.close();
179                    Block moved = ((BlockEntityMovingBlock) movingBlock).getMovingBlock();
180
181                    CompoundTag blockEntity = ((BlockEntityMovingBlock) movingBlock).getBlockEntity();
182
183                    if (blockEntity != null) {
184                        blockEntity.putInt("x", movingBlock.getFloorX());
185                        blockEntity.putInt("y", movingBlock.getFloorY());
186                        blockEntity.putInt("z", movingBlock.getFloorZ());
187                        BlockEntity.createBlockEntity(blockEntity.getString("id"), this.level.getChunk(movingBlock.getChunkX(), movingBlock.getChunkZ()), blockEntity);
188                    }
189
190                    this.level.setBlock(movingBlock, moved);
191                }
192            }
193
194            if (!extending && this.level.getBlock(getSide(facing)).getId() == (sticky? BlockID.PISTON_HEAD_STICKY : BlockID.PISTON_HEAD)) {
195                this.level.setBlock(getSide(facing), new BlockAir());
196                this.movable = true;
197            }
198
199            this.level.scheduleUpdate(this.getLevelBlock(), 1);
200            this.attachedBlocks.clear();
201            hasUpdate = false;
202        }
203
204        this.level.addChunkPacket(getChunkX(), getChunkZ(), getSpawnPacket());
205
206        return super.onUpdate() || hasUpdate;
207    }
208
209    private float getExtendedProgress(float progress) {
210        return this.extending ? progress - 1 : 1 - progress;
211    }
212
213    public boolean isBlockEntityValid() {
214        return true;
215    }
216
217    public void saveNBT() {
218        super.saveNBT();
219        this.namedTag.putByte("State", this.state);
220        this.namedTag.putByte("NewState", this.newState);
221        this.namedTag.putFloat("Progress", this.progress);
222        this.namedTag.putFloat("LastProgress", this.lastProgress);
223        this.namedTag.putBoolean("powered", this.powered);
224        this.namedTag.putList(getAttachedBlocks());
225        this.namedTag.putInt("facing", this.facing.getIndex());
226    }
227
228    public CompoundTag getSpawnCompound() {
229        return new CompoundTag()
230                .putString("id", BlockEntity.PISTON_ARM)
231                .putInt("x", (int) this.x)
232                .putInt("y", (int) this.y)
233                .putInt("z", (int) this.z)
234                .putFloat("Progress", this.progress)
235                .putFloat("LastProgress", this.lastProgress)
236                .putBoolean("isMovable", this.movable)
237                .putList(getAttachedBlocks())
238                .putList(new ListTag<>("BreakBlocks"))
239                .putBoolean("Sticky", this.sticky)
240                .putByte("State", this.state)
241                .putByte("NewState", this.newState);
242    }
243
244    private ListTag<IntTag> getAttachedBlocks() {
245        ListTag<IntTag> attachedBlocks = new ListTag<>("AttachedBlocks");
246        for (BlockVector3 block : this.attachedBlocks) {
247            attachedBlocks.add(new IntTag("", block.x));
248            attachedBlocks.add(new IntTag("", block.y));
249            attachedBlocks.add(new IntTag("", block.z));
250        }
251
252        return attachedBlocks;
253    }
