112    private static class JavaVersionRetriever implements Callable<Map<String, Map<String, Integer>>> {
113        // The following code can be attributed to the PaperMC project
114        // https://github.com/PaperMC/Paper/blob/master/Spigot-Server-Patches/0005-Paper-Metrics.patch#L614
115        @Override
116        public Map<String, Map<String, Integer>> call() {
117            Map<String, Map<String, Integer>> map = new HashMap<>();
118            String javaVersion = System.getProperty("java.version");
119            Map<String, Integer> entry = new HashMap<>();
120            entry.put(javaVersion, 1);
121            // http://openjdk.java.net/jeps/223
122            // Java decided to change their versioning scheme and in doing so modified the java.version system
123            // property to return $major[.$minor][.$secuity][-ea], as opposed to 1.$major.0_$identifier
124            // we can handle pre-9 by checking if the "major" is equal to "1", otherwise, 9+
125            String majorVersion = javaVersion.split("\\.")[0];
126            String release;
127            int indexOf = javaVersion.lastIndexOf('.');
128            if (majorVersion.equals("1")) {
129                release = "Java " + javaVersion.substring(0, indexOf);
130            } else {
131                // of course, it really wouldn't be all that simple if they didn't add a quirk, now would it
132                // valid strings for the major may potentially include values such as -ea to deannotate a pre release
133                Matcher versionMatcher = Pattern.compile("\\d+").matcher(majorVersion);
134                if (versionMatcher.find()) {
135                    majorVersion = versionMatcher.group(0);
136                }
137                release = "Java " + majorVersion;
138            }
139            map.put(release, entry);
140            return map;
141        }
142    }

147    private void loadConfig() throws IOException {
148        File bStatsFolder = new File(server.getPluginPath(), "bStats");
149        if (!bStatsFolder.exists() && !bStatsFolder.mkdirs()) {
150            log.warn("Failed to create bStats metrics directory");
151            return;
152        }
153        File configFile = new File(bStatsFolder, "config.yml");
154        if (!configFile.exists()) {
155            writeFile(configFile, "# bStats collects some data for plugin authors like how many servers are using their plugins.", "# To honor their work, you should not disable it.", "# This has nearly no effect on the server performance!", "# Check out https://bStats.org/ to learn more :)", "enabled: true", "serverUuid: \"" + UUID.randomUUID().toString() + "\"", "logFailedRequests: false");
156        }
157        Config config = new Config(configFile, Config.YAML);
158        // Load configuration
159        this.enabled = config.getBoolean("enabled", true);
160        this.serverUUID = config.getString("serverUuid");
161        this.logFailedRequests = config.getBoolean("logFailedRequests", false);
162    }

164    private void writeFile(File file, String... lines) throws IOException {
165        try (BufferedWriter writer = new BufferedWriter(new FileWriter(file))) {
166            for (String line : lines) {
167                writer.write(line);
168                writer.newLine();
169            }
170        }
