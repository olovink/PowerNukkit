019 */
020public class BlockTrapdoor extends BlockTransparentMeta implements Faceable {
021
022    public static final int TRAPDOOR_OPEN_BIT = 0x08;
023    public static final int TRAPDOOR_TOP_BIT = 0x04;
024
025    public BlockTrapdoor() {
026        this(0);
027    }
028
029    public BlockTrapdoor(int meta) {
030        super(meta);
031    }
032
033    @Override
034    public int getId() {
035        return TRAPDOOR;
036    }
037
038    @Override
039    public String getName() {
040        return "Oak Trapdoor";
041    }
042
043    @Override
044    public double getHardness() {
045        return 3;
046    }
047
048    @Override
049    public double getResistance() {
050        return 15;
051    }
052
053    @Override
054    public boolean canBeActivated() {
055        return true;
056    }
057
058    @Override
059    public int getToolType() {
060        return ItemTool.TYPE_AXE;
061    }
062
063    @Override
064    public int getWaterloggingLevel() {
065        return 1;
066    }
067
068    private static final AxisAlignedBB[] boundingBoxDamage = new AxisAlignedBB[16];
069
070    static {
071        for (int damage = 0; damage < 16; damage++) {
072            AxisAlignedBB bb;
073            double f = 0.1875;
074            if ((damage & TRAPDOOR_TOP_BIT) > 0) {
075                bb = new SimpleAxisAlignedBB(
076                        0,
077                        1 - f,
078                        0,
079                        1,
080                        1,
081                        1
082                );
083            } else {
084                bb = new SimpleAxisAlignedBB(
085                        0,
086                        0,
087                        0,
088                        1,
089                        0 + f,
090                        1
091                );
092            }
093            if ((damage & TRAPDOOR_OPEN_BIT) > 0) {
094                if ((damage & 0x03) == 0) {
095                    bb.setBounds(
096                            0,
097                            0,
098                            1 - f,
099                            1,
100                            1,
101                            1
102                    );
103                } else if ((damage & 0x03) == 1) {
104                    bb.setBounds(
105                            0,
106                            0,
107                            0,
108                            1,
109                            1,
110                            0 + f
111                    );
112                }
113                if ((damage & 0x03) == 2) {
114                    bb.setBounds(
115                            1 - f,
116                            0,
117                            0,
118                            1,
119                            1,
120                            1
121                    );
122                }
123                if ((damage & 0x03) == 3) {
124                    bb.setBounds(
125                            0,
126                            0,
127                            0,
128                            0 + f,
129                            1,
130                            1
131                    );
132                }
133            }
134            boundingBoxDamage[damage] = bb;
135        }
136    }
137
138    private AxisAlignedBB getRelativeBoundingBox() {
139        return boundingBoxDamage[this.getDamage()];
140    }
141

143    public double getMinX() {

160    }

166

171
172    @Override
173    public int onUpdate(int type) {
174        if (type == Level.BLOCK_UPDATE_REDSTONE && this.level.getServer().isRedstoneEnabled()) {
175            if ((!isOpen() && this.level.isBlockPowered(this.getLocation())) || (isOpen() && !this.level.isBlockPowered(this.getLocation()))) {
176                this.level.getServer().getPluginManager().callEvent(new BlockRedstoneEvent(this, isOpen() ? 15 : 0, isOpen() ? 0 : 15));
177                this.setDamage(this.getDamage() ^ 0x04);
178                this.level.setBlock(this, this, true);
179                this.level.addSound(this, isOpen() ? Sound.RANDOM_DOOR_OPEN : Sound.RANDOM_DOOR_CLOSE);
180                return type;
181            }
182        }
183
184        return 0;
185    }
186
187    @Override
188    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
189        BlockFace facing;
190        boolean top;
191        int meta = 0;
192
193        if (face.getAxis().isHorizontal() || player == null) {
194            facing = face;
195            top = fy > 0.5;
196        } else {
197            facing = player.getDirection().getOpposite();
198            top = face != BlockFace.UP;
199        }
200
201        int[] faces = {2, 1, 3, 0};
202        int faceBit = faces[facing.getHorizontalIndex()];
203        meta |= faceBit;
204
205        if (top) {
206            meta |= TRAPDOOR_TOP_BIT;
207        }
208        this.setDamage(meta);
209        this.getLevel().setBlock(block, this, true, true);
210        return true;
211    }
212
213    @Override
214    public Item toItem() {
215        return new ItemBlock(this, 0);
216    }
217
218    @Override
219    public boolean onActivate(Item item, Player player) {
220        if(toggle(player)) {
221            this.level.addSound(this, isOpen() ? Sound.RANDOM_DOOR_OPEN : Sound.RANDOM_DOOR_CLOSE);
222            return true;
223        }
224        return false;
225    }
226
227    public boolean toggle(Player player) {
228        DoorToggleEvent ev = new DoorToggleEvent(this, player);
229        getLevel().getServer().getPluginManager().callEvent(ev);
230        if(ev.isCancelled()) {
231            return false;
232        }
233        this.setDamage(this.getDamage() ^ TRAPDOOR_OPEN_BIT);
234        getLevel().setBlock(this, this, true);
235        return true;
236    }
237
238    @Override
239    public BlockColor getColor() {
240        return BlockColor.WOOD_BLOCK_COLOR;
241    }
242
243    public boolean isOpen() {
244        return (this.getDamage() & TRAPDOOR_OPEN_BIT) != 0;
245    }
246
247    public boolean isTop() {
248        return (this.getDamage() & TRAPDOOR_TOP_BIT) != 0;
249    }
250
251    @Override
252    public BlockFace getBlockFace() {
253        return BlockFace.fromHorizontalIndex(this.getDamage() & 0x07);
254    }
