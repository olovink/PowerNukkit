001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.blockentity.BlockEntity;
005import cn.nukkit.blockentity.BlockEntityBanner;
006import cn.nukkit.item.Item;
007import cn.nukkit.item.ItemTool;
008import cn.nukkit.level.Level;
009import cn.nukkit.math.AxisAlignedBB;
010import cn.nukkit.math.BlockFace;
011import cn.nukkit.math.NukkitMath;
012import cn.nukkit.nbt.tag.CompoundTag;
013import cn.nukkit.nbt.tag.IntTag;
014import cn.nukkit.nbt.tag.ListTag;
015import cn.nukkit.nbt.tag.Tag;
016import cn.nukkit.utils.BlockColor;
017import cn.nukkit.utils.DyeColor;
018import cn.nukkit.utils.Faceable;
019
020/**
021 * Created by PetteriM1
022 */
023public class BlockBanner extends BlockTransparentMeta implements Faceable {
024
025    public BlockBanner() {
026        this(0);
027    }
028
029    public BlockBanner(int meta) {
030        super(meta);
031    }
032
033    @Override
034    public int getId() {
035        return STANDING_BANNER;
036    }
037
038    @Override
039    public double getHardness() {
040        return 1;
041    }
042
043    @Override
044    public double getResistance() {
045        return 5;
046    }
047
048    @Override
049    public int getToolType() {
050        return ItemTool.TYPE_AXE;
051    }
052
053    @Override
054    public String getName() {
055        return "Banner";
056    }
057
058    @Override
059    protected AxisAlignedBB recalculateBoundingBox() {
060        return null;
061    }
062
063    @Override
064    public boolean canPassThrough() {
065        return true;
066    }
067
068    @Override
069    public int getWaterloggingLevel() {
070        return 1;
071    }
072
073    @Override
074    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
075        if (face != BlockFace.DOWN) {
076            if (face == BlockFace.UP) {
077                this.setDamage(NukkitMath.floorDouble(((player.yaw + 180) * 16 / 360) + 0.5) & 0x0f);
078                this.getLevel().setBlock(block, this, true);
079            } else {
080                this.setDamage(face.getIndex());
081                this.getLevel().setBlock(block, new BlockWallBanner(this.getDamage()), true);
082            }
083
084            CompoundTag nbt = BlockEntity.getDefaultCompound(this, BlockEntity.BANNER)
085                    .putInt("Base", item.getDamage() & 0xf);
086
087            Tag type = item.getNamedTagEntry("Type");
088            if (type instanceof IntTag) {
089                nbt.put("Type", type);
090            }
091            Tag patterns = item.getNamedTagEntry("Patterns");
092            if (patterns instanceof ListTag) {
093                nbt.put("Patterns", patterns);
094            }
095
096            BlockEntityBanner banner = (BlockEntityBanner) BlockEntity.createBlockEntity(BlockEntity.BANNER, this.getChunk(), nbt);
097            return banner != null;
098        }
099        return false;
100    }
101
102    @Override
103    public int onUpdate(int type) {
104        if (type == Level.BLOCK_UPDATE_NORMAL) {
105            if (this.down().getId() == Block.AIR) {
106                this.getLevel().useBreakOn(this);
107
108                return Level.BLOCK_UPDATE_NORMAL;
109            }
110        }
111
112        return 0;
113    }
114
115    @Override
116    public Item toItem() {
117        BlockEntity blockEntity = this.getLevel().getBlockEntity(this);
118        Item item = Item.get(Item.BANNER);
119        if (blockEntity instanceof BlockEntityBanner) {
120            BlockEntityBanner banner = (BlockEntityBanner) blockEntity;
121            item.setDamage(banner.getBaseColor() & 0xf);
122            item.setNamedTag((item.hasCompoundTag() ? item.getNamedTag() : new CompoundTag())
123                    .putInt("Base", banner.getBaseColor() & 0xf));
124            int type = banner.namedTag.getInt("Type");
125            if (type > 0) {
126                item.setNamedTag((item.hasCompoundTag() ? item.getNamedTag() : new CompoundTag())
127                        .putInt("Type", type));
128            }
129            ListTag<CompoundTag> patterns = banner.namedTag.getList("Patterns", CompoundTag.class);
130            if (patterns.size() > 0) {
131                item.setNamedTag((item.hasCompoundTag() ? item.getNamedTag() : new CompoundTag())
132                        .putList(patterns));
133            }
134        }
135        return item;
136    }
137
138    @Override
139    public BlockFace getBlockFace() {
140        return BlockFace.fromHorizontalIndex(this.getDamage() & 0x7);
141    }
142
143    @Override
144    public boolean breaksWhenMoved() {
145        return true;
146    }
147
148    @Override
149    public BlockColor getColor() {
150        return this.getDyeColor().getColor();
151    }
152
153    public DyeColor getDyeColor() {
154        if (this.level != null) {
155            BlockEntity blockEntity = this.level.getBlockEntity(this);
156
157            if (blockEntity instanceof BlockEntityBanner) {
158                return ((BlockEntityBanner) blockEntity).getDyeColor();
159            }
160        }
161
162        return DyeColor.WHITE;
163    }
164}
