001package cn.nukkit.block;
002
003import cn.nukkit.Player;
004import cn.nukkit.Server;
005import cn.nukkit.event.block.BlockGrowEvent;
006import cn.nukkit.item.Item;
007import cn.nukkit.item.ItemKelp;
008import cn.nukkit.level.Level;
009import cn.nukkit.level.particle.BoneMealParticle;
010import cn.nukkit.math.BlockFace;
011import cn.nukkit.math.MathHelper;
012
013import java.util.concurrent.ThreadLocalRandom;
014
015public class BlockKelp extends BlockFlowable {
016    public BlockKelp() {
017        this(0);
018    }
019    
020    public BlockKelp(int meta) {
021        super(meta);
022    }
023    
024    @Override
025    public int getId() {
026        return BLOCK_KELP;
027    }
028    
029    @Override
030    public String getName() {
031        return "Kelp";
032    }
033    
034    @Override
035    public boolean place(Item item, Block block, Block target, BlockFace face, double fx, double fy, double fz, Player player) {
036        Block down = down();
037        Block layer1Block = block.getLevelBlockAtLayer(1);
038        int waterDamage;
039        if ((down.getId() == BLOCK_KELP || down.isSolid()) && down.getId() != MAGMA && down.getId() != ICE && down.getId() != SOUL_SAND &&
040                (layer1Block instanceof BlockWater && ((waterDamage = (block.getDamage())) == 0 || waterDamage == 8))
041        ) {
042            if (waterDamage == 8) {
043                this.getLevel().setBlock(this, 1, new BlockWater(), true, false);
044            }
045            
046            if (down.getId() == BLOCK_KELP && down.getDamage() != 24) {
047                down.setDamage(24);
048                this.getLevel().setBlock(down, down, true, true);
049            }
050            
051            //Placing it by hand gives it a random age value between 0 and 24.
052            setDamage(ThreadLocalRandom.current().nextInt(25));
053            this.getLevel().setBlock(this, this, true, true);
054            return true;
055        } else {
056            return false;
057        }
058    }
059    
060    @Override
061    public int onUpdate(int type) {
062        if (type == Level.BLOCK_UPDATE_NORMAL) {
063            Block blockLayer1 = getLevelBlockAtLayer(1);
064            int waterDamage = 0;
065            if (!(blockLayer1 instanceof BlockIceFrosted) &&
066                    (!(blockLayer1 instanceof BlockWater) || ((waterDamage = blockLayer1.getDamage()) != 0 && waterDamage != 8))) {
067                this.getLevel().useBreakOn(this);
068                return type;
069            }
070        
071            Block down = down();
072            if ((!down.isSolid() && down.getId() != BLOCK_KELP) || down.getId() == MAGMA || down.getId() == ICE || down.getId() == SOUL_SAND) {
073                this.getLevel().useBreakOn(this);
074                return type;
075            }
076        
077            if (waterDamage == 8) {
078                this.getLevel().setBlock(this, 1, new BlockWater(), true, false);
079            }
080            return type;
081        } else if (type == Level.BLOCK_UPDATE_RANDOM) {
082            if (ThreadLocalRandom.current().nextInt(100) <= 14) {
083                grow();
084            }
085            return type;
086        }
087        return super.onUpdate(type);
088    }
089    
090    public boolean grow() {
091        int age = MathHelper.clamp(getDamage(), 0, 25);
092        if (age < 25) {
093            Block up = up();
094            if (up instanceof BlockWater && (up.getDamage() == 0 || up.getDamage() == 8)) {
095                BlockKelp grown = new BlockKelp(age + 1);
096                BlockGrowEvent ev = new BlockGrowEvent(this, grown);
097                Server.getInstance().getPluginManager().callEvent(ev);
098                if (!ev.isCancelled()) {
099                    this.setDamage(25);
100                    this.getLevel().setBlock(this, 0, this, true, true);
101                    this.getLevel().setBlock(up, 1, new BlockWater(), true, false);
102                    this.getLevel().setBlock(up, 0, ev.getNewState(), true, true);
103                    return true;
104                }
105            }
106        }
107        return false;
108    }
109    
110    @Override
111    public boolean onBreak(Item item) {
112        Block down = down();
113        if (down.getId() == BLOCK_KELP) {
114            this.getLevel().setBlock(down, new BlockKelp(ThreadLocalRandom.current().nextInt(25)), true, true);
115        }
116        this.getLevel().setBlock(this, new BlockAir(), true, true);
117        return true;
118    }
119    
120    @Override
121    public boolean onActivate(Item item, Player player) {
122        //Bone meal
123        if (item.getId() == Item.DYE && item.getDamage() == 0x0f) {
124            int x = (int) this.x;
125            int z = (int) this.z;
126            for (int y = (int) this.y + 1; y < 255; y++) {
127                int blockIdAbove = getLevel().getBlockIdAt(x, y, z);
128                if (blockIdAbove != BLOCK_KELP) {
129                    if (blockIdAbove == WATER || blockIdAbove == STILL_WATER) {
130                        int waterData = getLevel().getBlockDataAt(x, y, z);
131                        if (waterData == 0 || waterData == 8) {
132                            BlockKelp highestKelp = (BlockKelp) getLevel().getBlock(x, y - 1, z);
133                            if (highestKelp.grow()) {
134                                this.level.addParticle(new BoneMealParticle(this));
135    
136                                if (player != null && (player.gamemode & 0x01) == 0) {
137                                    item.count--;
138                                }
139                            }
140                        }
141                    }
142                    return false;
143                }
144            }
145            
146            return true;
147        }
148        
149        return false;
150    }
151    
152    @Override
153    public Item toItem() {
154        return new ItemKelp();
155    }
156    
157    @Override
158    public int getWaterloggingLevel() {
159        return 2;
160    }
161    
162    @Override
163    public boolean canBeActivated() {
164        return true;
165    }
166}
