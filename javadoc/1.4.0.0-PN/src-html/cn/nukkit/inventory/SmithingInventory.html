019package cn.nukkit.inventory;
020
021import cn.nukkit.Player;
022import cn.nukkit.Server;
023import cn.nukkit.api.PowerNukkitOnly;
024import cn.nukkit.api.Since;
025import cn.nukkit.item.Item;
026import cn.nukkit.level.Position;
027
028import javax.annotation.Nonnull;
029import javax.annotation.Nullable;
030
031/**
032 * @author joserobjr
033 * @since 2020-09-28
034 */
035@PowerNukkitOnly
036@Since("1.4.0.0-PN")
037public class SmithingInventory extends FakeBlockUIComponent {
038    private static final int EQUIPMENT = 0;
039    private static final int INGREDIENT = 1;
040    
041    @PowerNukkitOnly
042    @Since("1.4.0.0-PN")
043    public static final int SMITHING_EQUIPMENT_UI_SLOT = 51;
044
045    @PowerNukkitOnly
046    @Since("1.4.0.0-PN")
047    public static final int SMITHING_INGREDIENT_UI_SLOT = 52;
048
049    @PowerNukkitOnly
050    @Since("1.4.0.0-PN")
051    public SmithingInventory(PlayerUIInventory playerUI, Position position) {
052        super(playerUI, InventoryType.SMITHING_TABLE, 51, position);
053    }
054
055    @PowerNukkitOnly
056    @Since("1.4.0.0-PN")
057    @Nullable
058    public SmithingRecipe matchRecipe() {
059        return Server.getInstance().getCraftingManager().matchSmithingRecipe(getEquipment(), getIngredient());
060    }
061
062    @Override
063    public void onSlotChange(int index, Item before, boolean send) {
064        if (index == EQUIPMENT || index == INGREDIENT) {
065            updateResult();
066        }
067        super.onSlotChange(index, before, send);
068    }
069
070    @PowerNukkitOnly
071    @Since("1.4.0.0-PN")
072    public void updateResult() {
073        Item result;
074        SmithingRecipe recipe = matchRecipe();
075        if (recipe == null) {
076            result =  Item.get(0);
077        } else {
078            result = recipe.getFinalResult(getEquipment());
079        }
080        setResult(result);
081    }
082    
083    private void setResult(Item result) {
084        //playerUI.setItem(50, result);
085    }
086
087    @PowerNukkitOnly
088    @Since("1.4.0.0-PN")
089    @Nonnull
090    public Item getResult() {
091        SmithingRecipe recipe = matchRecipe();
092        if (recipe == null) {
093            return Item.get(0);
094        }
095        return recipe.getFinalResult(getEquipment());
096    }
097
098    @PowerNukkitOnly
099    @Since("1.4.0.0-PN")
100    public Item getEquipment() {
101        return getItem(EQUIPMENT);
102    }
103
104    @PowerNukkitOnly
105    @Since("1.4.0.0-PN")
106    public void setEquipment(Item equipment) {
107        setItem(EQUIPMENT, equipment);
108    }
109
110    @PowerNukkitOnly
111    @Since("1.4.0.0-PN")
112    public Item getIngredient() {
113        return getItem(INGREDIENT);
114    }
115
116    @PowerNukkitOnly
117    @Since("1.4.0.0-PN")
118    public void setIngredient(Item ingredient) {
119        setItem(INGREDIENT, ingredient);
120    }
121
122    @Override
123    public void onOpen(Player who) {
124        super.onOpen(who);
125        who.craftingType = Player.CRAFTING_SMITHING;
126    }
127
128    @Override
129    public void onClose(Player who) {
130        super.onClose(who);
131        who.craftingType = Player.CRAFTING_SMALL;
132
133        who.giveItem(getItem(EQUIPMENT), getItem(INGREDIENT));
134        
135        this.clear(EQUIPMENT);
136        this.clear(INGREDIENT);
137        playerUI.clear(50);
138    }
139}
