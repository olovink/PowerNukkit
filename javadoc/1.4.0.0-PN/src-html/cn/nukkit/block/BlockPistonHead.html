001package cn.nukkit.block;
002
003import cn.nukkit.api.PowerNukkitDifference;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.item.Item;
007import cn.nukkit.item.ItemBlock;
008import cn.nukkit.math.BlockFace;
009import cn.nukkit.utils.Faceable;
010
011/**
012 * @author CreeperFace
013 */
014public class BlockPistonHead extends BlockTransparentMeta implements Faceable {
015
016    public BlockPistonHead() {
017        this(0);
018    }
019
020    public BlockPistonHead(int meta) {
021        super(meta);
022    }
023
024    @Override
025    public int getId() {
026        return PISTON_HEAD;
027    }
028
029    @Override
030    public String getName() {
031        return "Piston Head";
032    }
033
034    @Override
035    public double getResistance() {
036        return 2.5;
037    }
038
039    @Override
040    public double getHardness() {
041        return 0.5;
042    }
043
044    @Override
045    public Item[] getDrops(Item item) {
046        return Item.EMPTY_ARRAY;
047    }
048
049    @Override
050    @PowerNukkitDifference(info = "Remove BlockEntity of piston on break.", since = "1.4.0.0-PN")
051    public boolean onBreak(Item item) {
052        this.level.setBlock(this, Block.get(BlockID.AIR), true, true);
053        Block side = getSide(getBlockFace().getOpposite());
054
055        if (side instanceof BlockPistonBase && ((BlockPistonBase) side).getBlockFace() == this.getBlockFace()) {
056            BlockPistonBase piston = (BlockPistonBase) side;
057            piston.onBreak(item);
058
059            if(piston.getBlockEntity() != null) piston.getBlockEntity().close();
060        }
061        return true;
062    }
063
064    @Override
065    public BlockFace getBlockFace() {
066        BlockFace face = BlockFace.fromIndex(this.getDamage());
067
068        return face.getHorizontalIndex() >= 0 ? face.getOpposite() : face;
069    }
070
071    @Override
072    public boolean canBePushed() {
073        return false;
074    }
075
076    @Override
077    public boolean canBePulled() {
078        return false;
079    }
080
081    @Override
082    public boolean isSolid() {
083        return false;
084    }
085
086    @Since("1.3.0.0-PN")
087    @PowerNukkitOnly
088    @Override
089    public boolean isSolid(BlockFace side) {
090        return false;
091    }
092
093    @Override
094    public Item toItem() {
095        return new ItemBlock(Block.get(BlockID.AIR));
096    }
097}
