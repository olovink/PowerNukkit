001package cn.nukkit.block;
002
003import cn.nukkit.api.PowerNukkitOnly;
004import cn.nukkit.api.Since;
005import cn.nukkit.blockproperty.BlockProperties;
006import cn.nukkit.item.Item;
007import cn.nukkit.item.ItemTool;
008import cn.nukkit.utils.BlockColor;
009
010import javax.annotation.Nonnull;
011
012@PowerNukkitOnly
013@Since("1.4.0.0-PN")
014public class BlockSlabBlackstone extends BlockSlab {
015    @PowerNukkitOnly
016    @Since("1.4.0.0-PN")
017    public BlockSlabBlackstone() {
018        this(0);
019    }
020
021    @PowerNukkitOnly
022    @Since("1.4.0.0-PN")
023    public BlockSlabBlackstone(int meta) {
024        super(meta, BLACKSTONE_DOUBLE_SLAB);
025    }
026
027    @Override
028    public int getId() {
029        return BLACKSTONE_SLAB;
030    }
031
032    @Since("1.4.0.0-PN")
033    @PowerNukkitOnly
034    @Nonnull
035    @Override
036    public BlockProperties getProperties() {
037        return SIMPLE_SLAB_PROPERTIES;
038    }
039
040    @Override
041    public String getSlabName() {
042        return "Blackstone Slab";
043    }
044
045    @Override
046    public boolean isSameType(BlockSlab slab) {
047        return slab.getId() == getId();
048    }
049
050    @Override
051    public int getToolType() {
052        return ItemTool.TYPE_PICKAXE;
053    }
054
055    @Override
056    public double getResistance() {
057        return 6;
058    }
059
060    @Override
061    public Item[] getDrops(Item item) {
062        if (item.isPickaxe() && item.getTier() >= ItemTool.TIER_WOODEN) {
063            return new Item[]{
064                    toItem()
065            };
066        } else {
067            return Item.EMPTY_ARRAY;
068        }
069    }
070
071    @Override
072    public BlockColor getColor() {
073        return BlockColor.BLACK_BLOCK_COLOR;
074    }
075
076    @Override
077    public boolean canHarvestWithHand() {
078        return false;
079    }
080}
