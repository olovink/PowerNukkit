001package cn.nukkit.inventory;
002
003import cn.nukkit.Player;
004import cn.nukkit.api.PowerNukkitOnly;
005import cn.nukkit.api.Since;
006import cn.nukkit.item.Item;
007
008import java.util.Collections;
009import java.util.Map;
010import java.util.Set;
011
012public class PlayerUIComponent extends BaseInventory {
013
014    @Since("1.4.0.0-PN") public static final int CREATED_ITEM_OUTPUT_UI_SLOT = 50;
015
016    @PowerNukkitOnly protected final PlayerUIInventory playerUI;
017    private final int offset;
018    private final int size;
019
020    PlayerUIComponent(PlayerUIInventory playerUI, int offset, int size) {
021        super(playerUI.holder, InventoryType.UI, Collections.emptyMap(), size);
022        this.playerUI = playerUI;
023        this.offset = offset;
024        this.size = size;
025    }
026
027    @Override
028    public int getSize() {
029        return size;
030    }
031
032    @Override
033    public int getMaxStackSize() {
034        return 64;
035    }
036
037    @Override
038    public void setMaxStackSize(int size) {
039        throw new UnsupportedOperationException();
040    }
041
042
043    @Override
044    public String getTitle() {
045        throw new UnsupportedOperationException();
046    }
047
048    @Override
049    public Item getItem(int index) {
050        return this.playerUI.getItem(index + this.offset);
051    }
052
053    @Override
054    public boolean setItem(int index, Item item, boolean send) {
055        Item before = playerUI.getItem(index + this.offset);
056        if (this.playerUI.setItem(index + this.offset, item, send)) {
057            onSlotChange(index, before, false);
058            return true;
059        }
060        return false;
061    }
062
063    @Override
064    public boolean clear(int index, boolean send) {
065        Item before = playerUI.getItem(index + this.offset);
066        if (this.playerUI.clear(index + this.offset, send)) {
067            onSlotChange(index, before, false);
068            return true;
069        }
070        return false;
071    }
072
073    @Override
074    public Map<Integer, Item> getContents() {
075        Map<Integer, Item> contents = playerUI.getContents();
076        contents.keySet().removeIf(slot -> slot < offset || slot > offset + size);
077        return contents;
078    }
079
080
081    @Override
082    public void sendContents(Player... players) {
083        this.playerUI.sendContents(players);
084    }
085
086    @Override
087    public void sendSlot(int index, Player... players) {
088        playerUI.sendSlot(index + this.offset, players);
089    }
090
091    @Override
092    public Set<Player> getViewers() {
093        return playerUI.getViewers();
094    }
095
096    @Override
097    public InventoryType getType() {
098        return playerUI.type;
099    }
100
101    @Override
102    public void onOpen(Player who) {
103        
104    }
105
106    @Override
107    public boolean open(Player who) {
108        return false;
109    }
110
111    @Override
112    public void close(Player who) {
113
114    }
115
116    @Override
117    public void onClose(Player who) {
118
119    }
120
121    @Override
122    public void onSlotChange(int index, Item before, boolean send) {
123        if (send) {
124            this.playerUI.onSlotChangeBase(index + this.offset, before, true);
125        }
126        super.onSlotChange(index, before, false);
127    }
128}
