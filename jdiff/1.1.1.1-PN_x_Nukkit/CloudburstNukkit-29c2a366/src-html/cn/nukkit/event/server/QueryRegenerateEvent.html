052    public QueryRegenerateEvent(Server server, int timeout) {
053        this.timeout = timeout;
054        this.serverName = server.getMotd();
055        this.listPlugins = server.getConfig("settings.query-plugins", true);
056        this.plugins = server.getPluginManager().getPlugins().values().toArray(new Plugin[0]);
057        this.players = server.getOnlinePlayers().values().toArray(new Player[0]);
058        this.gameType = (server.getGamemode() & 0x01) == 0 ? "SMP" : "CMP";
059        this.version = server.getVersion();
060        this.server_engine = server.getName() + " " + server.getNukkitVersion();
061        this.map = server.getDefaultLevel() == null ? "unknown" : server.getDefaultLevel().getName();
062        this.numPlayers = this.players.length;
063        this.maxPlayers = server.getMaxPlayers();
064        this.whitelist = server.hasWhitelist() ? "on" : "off";
065        this.port = server.getPort();
066        this.ip = server.getIp();
067    }

141    public byte[] getLongQuery(byte[] buffer) {
142        if (buffer == null) buffer = new byte[Character.MAX_VALUE];
143        FastByteArrayOutputStream query = new FastByteArrayOutputStream(buffer);
144        try {
145            String plist = this.server_engine;
146            if (this.plugins.length > 0 && this.listPlugins) {
147                plist += ":";
148                for (Plugin p : this.plugins) {
149                    PluginDescription d = p.getDescription();
150                    plist += " " + d.getName().replace(";", "").replace(":", "").replace(" ", "_") + " " + d.getVersion().replace(";", "").replace(":", "").replace(" ", "_") + ";";
151                }
152                plist = plist.substring(0, plist.length() - 2);
153            }
154
155            query.write("splitnum".getBytes());
156            query.write((byte) 0x00);
157            query.write((byte) 128);
158            query.write((byte) 0x00);
159
160            LinkedHashMap<String, String> KVdata = new LinkedHashMap<>();
161            KVdata.put("hostname", this.serverName);
162            KVdata.put("gametype", this.gameType);
163            KVdata.put("game_id", GAME_ID);
164            KVdata.put("version", this.version);
165            KVdata.put("server_engine", this.server_engine);
166            KVdata.put("plugins", plist);
167            KVdata.put("map", this.map);
168            KVdata.put("numplayers", String.valueOf(this.numPlayers));
169            KVdata.put("maxplayers", String.valueOf(this.maxPlayers));
170            KVdata.put("whitelist", this.whitelist);
171            KVdata.put("hostip", this.ip);
172            KVdata.put("hostport", String.valueOf(this.port));
173
174            for (Map.Entry<String, String> entry : KVdata.entrySet()) {
175                query.write(entry.getKey().getBytes(StandardCharsets.UTF_8));
176                query.write((byte) 0x00);
177                query.write(entry.getValue().getBytes(StandardCharsets.UTF_8));
178                query.write((byte) 0x00);
179            }
180
181            query.write(new byte[]{0x00, 0x01});
182            query.write("player_".getBytes());
183            query.write(new byte[]{0x00, 0x00});
184
185            for (Player player : this.players) {
186                query.write(player.getName().getBytes(StandardCharsets.UTF_8));
187                query.write((byte) 0x00);
188            }
189
190            query.write((byte) 0x00);
191        } catch (IOException e) {
192            e.printStackTrace();
193        }
194        return query.toByteArray();
195    }

197    public byte[] getShortQuery(byte[] buffer) {
198        if (buffer == null) buffer = new byte[Character.MAX_VALUE];
199        FastByteArrayOutputStream query = new FastByteArrayOutputStream(buffer);
200        try {
201            query.write(this.serverName.getBytes(StandardCharsets.UTF_8));
202            query.write((byte) 0x00);
203            query.write(this.gameType.getBytes(StandardCharsets.UTF_8));
204            query.write((byte) 0x00);
205            query.write(this.map.getBytes(StandardCharsets.UTF_8));
206            query.write((byte) 0x00);
207            query.write(String.valueOf(this.numPlayers).getBytes(StandardCharsets.UTF_8));
208            query.write((byte) 0x00);
209            query.write(String.valueOf(this.maxPlayers).getBytes(StandardCharsets.UTF_8));
210            query.write((byte) 0x00);
211            query.write(Binary.writeLShort(this.port));
212            query.write(this.ip.getBytes(StandardCharsets.UTF_8));
213            query.write((byte) 0x00);
214        } catch (IOException e) {
215            e.printStackTrace();
216        }
217        return query.toByteArray();
218    }
