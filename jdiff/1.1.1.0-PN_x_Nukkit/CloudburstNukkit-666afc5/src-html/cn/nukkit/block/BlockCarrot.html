001package cn.nukkit.block;
002
003import cn.nukkit.item.Item;
004import cn.nukkit.item.ItemCarrot;
005
006import java.util.Random;
007
008/**
009 * @author Nukkit Project Team
010 */
011public class BlockCarrot extends BlockCrops {
012
013    public BlockCarrot(int meta) {
014        super(meta);
015    }
016
017    public BlockCarrot() {
018        this(0);
019    }
020
021    @Override
022    public String getName() {
023        return "Carrot Block";
024    }
025
026    @Override
027    public int getId() {
028        return CARROT_BLOCK;
029    }
030
031    @Override
032    public Item[] getDrops(Item item) {
033        if (getDamage() >= 0x07) {
034            return new Item[]{
035                    new ItemCarrot(0, new Random().nextInt(3) + 1)
036            };
037        }
038        return new Item[]{
039                new ItemCarrot()
040        };
041    }
042
043    @Override
044    public Item toItem() {
045        return new ItemCarrot();
046    }
047}
